$$
\delta(x)
\quad=\quad
\begin{cases} 
          0, & x \neq 0\\
          \infty, & x=0
\end{cases} 
\quad=\quad \lim_{ a \to 0 } \frac{1}{|a| \sqrt{ \pi }}e^{-(x/a)^2} 
$$
$$
\int^{\infty}_{-\infty} \delta(x)dx = 1
$$
$$
\int^{\infty}_{-\infty} \delta(t-a)f(t) dt = f(a)
$$
